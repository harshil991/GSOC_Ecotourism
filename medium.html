platypus_taxon <- search_taxa("Ornithorhynchus anatinus")

platypus_data <- galah_call() %>%
  galah_identify(platypus_taxon) %>%
  galah_filter(year == 2024) %>%
  galah_select("decimalLatitude", "decimalLongitude") %>%
  atlas_occurrences()

## Downloading
platypus_sf <- platypus_data %>%
  filter(!is.na(decimalLatitude) & !is.na(decimalLongitude)) %>%
  st_as_sf(coords = c("decimalLongitude", "decimalLatitude"), crs = 4326)
load(system.file("extdata", "isd_history.rda", package = "GSODR"))

stations_sf <- isd_history %>%
  filter(COUNTRY_NAME == "AUSTRALIA") %>%
  st_as_sf(coords = c("LON", "LAT"), crs = 4326)

victoria_sf <- ozmaps::ozmap_states %>%
  filter(NAME == "Victoria") %>%
  st_transform(crs = 4326)
stations_in_vic <- st_intersection(stations_sf, victoria_sf)
## Warning: attribute variables are assumed to be spatially constant throughout
## all geometries
distance_matrix <- distm(
  st_coordinates(platypus_sf),
  st_coordinates(stations_in_vic)
)
nearest_station_idx <- apply(distance_matrix, 1, which.min)

platypus_stations <- platypus_sf %>%
  mutate(
    station_id = stations_in_vic$STNID[nearest_station_idx],
    station_name = stations_in_vic$NAME[nearest_station_idx]
  )
weather_data <- GSODR::get_GSOD(years = 2024, station = unique(platypus_stations$station_id))

library(ggplot2)

ggplot() +
  geom_sf(data = victoria_sf, fill = "white", color = "black") +
  geom_sf(data = stations_in_vic, shape = 21, fill = "skyblue", color = "blue", size = 2, alpha = 0.6) +
  geom_sf(data = platypus_sf, aes(), shape = 21, fill = "darkgreen", color = "black", size = 2, alpha = 0.7) +
  geom_sf(data = platypus_stations, aes(geometry = geometry), color = "red", shape = 4, size = 1.5) +
  theme_minimal() +
  labs(
    title = "Platypus Sightings (2024) and Nearby Weather Stations in Victoria",
    subtitle = "Matching each sighting with the nearest GSOD weather station",
    caption = "Data: Atlas of Living Australia & GSODR"
  ) +
  theme(
    plot.title = element_text(size = 14, face = "bold"),
    plot.subtitle = element_text(size = 10),
    legend.position = "none"
  )
